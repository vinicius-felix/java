        private double saldo;

        // Construtor Cheio
        Funcionario(String nome, String cpf, String telefone, int conta, double saldo) {
          super(nome, cpf, telefone);
          this.conta = conta;
          this.saldo = saldo;

      }
    
  
    
      // Classe Principal
      public class Principal {
        public static void main(String[] args) {
          
          // Criando uma lista do tipo Funcionario
          ArrayList< Funcionario > funcs = new ArrayList< Funcionario >();

          // Adicionando um objeto do tipo Funcionario na lista
          funcs.add(new Funcionario("Nome", "000.111.222-33", "99 9 1111-9999", 898989, 100000.99));

          // Exibindos os dados da lista
          for(int i = 0; i < funcs.size(); i++) { 
            System.out.printf("Dados Cliente %d", i+1); System.out.println("Nome: " + funcs.get(i).getNome());
            System.out.println(" Cpf: " + funcs.get(i).getCpf());
            System.out.println(" Telefone: " + funcs.get(i).getTelefone());
            System.out.println(" Conta: " + funcs.get(i).getConta());
            System.out.println(" Saldo: " + funcs.get(i).getSaldo());
